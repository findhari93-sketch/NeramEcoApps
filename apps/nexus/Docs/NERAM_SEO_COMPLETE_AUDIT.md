### Fix 1: XML Sitemap — neramclasses.com

**Verification command:**
```bash
curl -s -o /dev/null -w "%{http_code}" https://neramclasses.com/sitemap.xml
# Expected: 200
# If 404: sitemap is not deployed
```

**If not deployed, implement:**

```bash
cd apps/neramclasses  # adjust path to main site app
npm install next-sitemap
```

Create `next-sitemap.config.js` at the app root:
```js
/** @type {import('next-sitemap').IConfig} */
module.exports = {
  siteUrl: 'https://neramclasses.com',
  generateRobotsTxt: true,
  sitemapSize: 5000,
  changefreq: 'weekly',
  priority: 0.7,
  exclude: [
    '/api/*',
    '/admin/*',
    '/dashboard/*',
    '/en/*',
    '/inner-page',
    '/NATA_Coaching_center_near_me_address',
    '/JEE_B.arch_Syllabus',
    '/nata-coaching-online',  // duplicate of /best-nata-coaching-online
  ],
  robotsTxtOptions: {
    policies: [
      {
        userAgent: '*',
        allow: '/',
        disallow: ['/api/', '/admin/', '/dashboard/', '/en/', '/_next/'],
      },
    ],
  },
  transform: async (config, path) => {
    if (path === '/') {
      return { loc: path, changefreq: 'daily', priority: 1.0, lastmod: new Date().toISOString() };
    }
    if (path.startsWith('/courses/')) {
      return { loc: path, changefreq: 'weekly', priority: 0.9, lastmod: new Date().toISOString() };
    }
    if (path.includes('cutoff') || path.includes('predictor') || path.includes('calculator')) {
      return { loc: path, changefreq: 'monthly', priority: 0.8, lastmod: new Date().toISOString() };
    }
    if (path.startsWith('/coaching/') || path.startsWith('/contact/nata-coaching-center')) {
      return { loc: path, changefreq: 'monthly', priority: 0.7, lastmod: new Date().toISOString() };
    }
    if (path.startsWith('/blog/')) {
      return { loc: path, changefreq: 'monthly', priority: 0.6, lastmod: new Date().toISOString() };
    }
    return { loc: path, changefreq: config.changefreq, priority: config.priority, lastmod: new Date().toISOString() };
  },
};
```

Add to `package.json`:
```json
{
  "scripts": {
    "build": "next build",
    "postbuild": "next-sitemap"
  }
}
```

Build and verify:
```bash
npm run build
cat public/sitemap-0.xml | head -50
cat public/robots.txt
```

```js
module.exports = {

  generateRobotsTxt: true,
  exclude: [
    '/api/*', '/dashboard/*', '/auth/*', '/login', '/signup', '/profile/*', '/settings/*',
  ],
  robotsTxtOptions: {
    policies: [
      {
        userAgent: '*',
        allow: ['/tools', '/tools/*'],
        disallow: ['/api/', '/dashboard/', '/auth/', '/login', '/signup', '/profile/', '/settings/', '/_next/'],
      },
    ],
  },
  transform: async (config, path) => {
    if (path === '/tools' || path.startsWith('/tools/')) {
      return { loc: path, changefreq: 'weekly', priority: path === '/tools' ? 0.9 : 0.8, lastmod: new Date().toISOString() };
    }
    return null; // exclude everything else from sitemap
  },
};
```
